namespace App\Parsers;

class SafariPackageParser
{
    private $content;
    private $packages = [];

    public function __construct($filePath)
    {
        $this->content = file_get_contents($filePath);
        $this->parse();
    }

    public function getPackages()
    {
        return $this->packages;
    }

    private function parse()
    {
        // Split the content into sections for each package
        // The file seems to have packages separated by multiple blank lines
        $sections = preg_split('/\n\s*\n\s*\n/', $this->content);
        
        foreach ($sections as $section) {
            if (trim($section) === '') {
                continue;
            }
            
            // Check if this section looks like a package (contains "Nights / Days")
            if (preg_match('/(\d+)\s*Nights\s*\/\s*(\d+)\s*Days/', $section, $durationMatches)) {
                $package = $this->parsePackage($section, $durationMatches);
                if ($package) {
                    $this->packages[] = $package;
                }
            }
        }
    }

    private function parsePackage($section, $durationMatches)
    {
        $package = [
            'title' => '',
            'duration_nights' => (int)$durationMatches[1],
            'duration_days' => (int)$durationMatches[2],
            'activities' => [],
            'itinerary' => [],
            'costings' => [],
            'type' => $this->determinePackageType($section),

        // Extract title (first line before duration)
        $lines = explode("\n", $section);
        $package['title'] = trim($lines[0]);

        // Extract activities
        if (preg_match('/Activities\s*\/\s*Visit:\s*(.+?)(?=Day \d+:)/s', $section, $activitiesMatch)) {
